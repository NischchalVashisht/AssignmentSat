def area(shape:String,first:Int,second:Int,f:(Int,Int)=>Int):String={

  shape match{
    case "rectangle"=>"Area of rectangle is"+f(first,second)
    case "rhombus"=>"Area of rhombus is"+f(first,second)
    case "parallelogram"=>"Area of parallelogram is"+f(first,second)

    case  _ =>s"Not define yet for $shape"
  }
}

def rectangle(x:Int,y:Int):Int=
{

  x * y
}

def rhombus(x:Int,y:Int):Int=
{
  (x * y)/2
}

def parallelogram(x:Int,y:Int):Int=
{

  x * y
}

area("rectangle",10,3,rectangle)
